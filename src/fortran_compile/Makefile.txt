# Fortran compiler
F77 = gfortran-mp-14

# Flags
OPT = -g

# All .F sources
SRCS = $(wildcard *.F)

# Convert program.F → program.xgf
EXECUTABLES = $(SRCS:.F=.xgf)

# Default: build all executables
all: $(EXECUTABLES)

# Rule: program.F → program.xgf
%.xgf: %.F
	$(F77) $(OPT) $< -o $@

# Clean everything
clean:
	rm -f *.xgf
